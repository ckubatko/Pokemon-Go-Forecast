    .city-card h2 { margin: 0 0 10px; font-size: 1.2em; }

<body>
  <h1>Pokémon GO Weather – U.S. Cities</h1>
  <div id="weather-container"></div>

  <script>
    const cities = [
      "Seattle", "Los Angeles", "Las Vegas", 
      "New York", "Chicago", "Miami", 
      "Houston", "Boston"
    ];

    const apiKey = "YOUR_API_KEY_HERE"; // Replace this with your real API key

    const boostedTypes = {
      Clear: ["Fire", "Grass", "Ground"],
      Clouds: ["Normal", "Flying"],
      Rain: ["Water", "Bug", "Electric"],
      Snow: ["Ice", "Steel"],
      Thunderstorm: ["Electric", "Dark"],
      Drizzle: ["Water"],
      Mist: ["Ghost"],
      Fog: ["Ghost"],
      Haze: ["Fairy"],
      Smoke: ["Poison"],
      Dust: ["Ground", "Rock"],
      Sand: ["Ground", "Rock"],
      Ash: ["Fire"],
      Squall: ["Flying"],
      Tornado: ["Flying"]
    };

    const container = document.getElementById("weather-container");

    cities.forEach(city => {
      fetch(`https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${apiKey}&units=imperial`)
        .then(res => res.json())
        .then(data => {
          const condition = data.weather[0].main;
          const iconCode = data.weather[0].icon;
          const temp = Math.round(data.main.temp);
          const boost = boostedTypes[condition] || ["Unknown"];

          const card = document.createElement("div");
          card.className = "city-card";
          card.innerHTML = `
            <h2>${city}</h2>
            <img class="weather-icon" src="https://openweathermap.org/img/wn/${iconCode}@2x.png" alt="${condition}">
            <p><strong>${condition}</strong></p>
            <p>Temp: ${temp}°F</p>
            <p class="boosted-types">Boosted Types: ${boost.join(", ")}</p>
          `;
          container.appendChild(card);
        })
        .catch(err => {
          console.error(`Failed to load weather for ${city}:`, err);
        });
    });
  </script>
